version: '3'

services:

  apache:
    image: php:7.4-apache
    ports:
      - 8080:80
    volumes:
      - .:/var/www/html
    command: '/bin/sh -c "docker-php-ext-install mysqli && exec apache2-foreground"'
    restart: 'always'

  mysql:
    image: mysql:latest
    restart: 'always'
    volumes:
      - ./init.sql:/init/init.sql
    ports: 
      - 3306:3306
    command: --init-file /init/init.sql
    environment:
      MYSQL_USER : user
      MYSQL_ROOT_PASSWORD: password
      MYSQL_DATABASE: appDB
